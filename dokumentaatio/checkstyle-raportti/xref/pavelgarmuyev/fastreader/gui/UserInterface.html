9   public class UserInterface implements Runnable {

11      private JFrame frame;
12      private Commands commands;
13  
14      public UserInterface(Commands commands) {
15          this.commands = commands;
16      }
17  
18      @Override
19      public void run() {
20          frame = new JFrame("FastReader");
21          frame.setPreferredSize(new Dimension(500, 300));
22  
23          frame.getContentPane().setBackground(Color.WHITE);
24  
25          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
26          frame.setResizable(false);
27          frame.setJMenuBar(createMenuBar());
28          createComponents(frame.getContentPane());
29  
30          frame.pack();
31          Dimension dimension = Toolkit.getDefaultToolkit().getScreenSize();
32          frame.setLocation((int)(dimension.getWidth() - frame.getWidth()) / 2,
33                          (int)(dimension.getHeight() - frame.getHeight()) / 2);
34          frame.setVisible(true);
35  
36      }
37  
38      private void createComponents(Container container) {
39          container.setLayout(new BoxLayout(container, BoxLayout.Y_AXIS));
40  
41          container.add(createBigWord());
42          container.add(createControls());
43          container.add(Box.createVerticalGlue());
44  
45          container.add(Box.createVerticalGlue());
46          container.add(createProgressBar());
47      }
48  
49      private JProgressBar createProgressBar() {
50          JProgressBar progressBar = new JProgressBar(0, commands.getTotalWords());
51          progressBar.setValue(0);
52          progressBar.setStringPainted(true);
53          progressBar.setAlignmentX(0.9f);
54  
55          return progressBar;
56      }
57  
58      private JPanel createBigWord() {
59  
60          JPanel bigWordPanel = new JPanel();
61          bigWordPanel.setLayout(new BoxLayout(bigWordPanel, BoxLayout.X_AXIS));
62  
63          JLabel bigWord = new JLabel("text", SwingConstants.CENTER);
64          bigWord.setAlignmentX(0.5f);
65          bigWord.setBorder(BorderFactory.createMatteBorder(2, 0, 2, 0, Color.BLACK));
66          bigWord.setMinimumSize(new Dimension(300, 100));
67          bigWord.setMaximumSize(new Dimension(300, 100));
68          bigWord.setFont(new Font("sans-serif", Font.PLAIN, 50));
69          bigWord.setBackground(Color.WHITE);
70          bigWord.setOpaque(true);
71  
72          bigWordPanel.add(bigWord);
73          return bigWordPanel;
74      }
75  
76      private JPanel createControls() {
77  
78          JPanel controlsPanel = new JPanel();
79          controlsPanel.setLayout(new BoxLayout(controlsPanel, BoxLayout.X_AXIS));
80          controlsPanel.setBackground(Color.WHITE);
81  
82          LinkedList<JButton> buttonsList = new LinkedList<>();
83  
84          JButton rewindButton = new JButton(new ImageIcon("rewind_button.png"));
85          rewindButton.addActionListener(new RewindActionListener(commands));
86          buttonsList.add(rewindButton);
87  
88          JButton prevButton = new JButton(new ImageIcon("prev_sentence_button.png"));
89          prevButton.addActionListener(new BackwardsActionListener(commands));
90          buttonsList.add(prevButton);
91  
92          JButton playButton = new JButton(new ImageIcon("play_button.png"));
93          playButton.addActionListener(new PlayActionListener());
94          buttonsList.add(playButton);
95  
96          JButton nextButton = new JButton(new ImageIcon("next_sentence_button.png"));
97          nextButton.addActionListener(new ForwardsActionListener(commands));
98          buttonsList.add(nextButton);
99  
100         for (JButton jb : buttonsList) {
101             jb.setOpaque(false);
102             jb.setContentAreaFilled(false);
103             jb.setBorderPainted(false);
104             controlsPanel.add(jb);
105         }
106         return controlsPanel;
107     }
108

122         JMenu recentSubMenu = new JMenu("Open recent");
123 
124         JMenuItem recentFile = new JMenuItem("/path/to/file.txt");
125 
126         recentSubMenu.add(recentFile);
127         fileMenu.add(recentSubMenu);
128 
129         return menuBar;
130     }
