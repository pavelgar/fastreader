6   import java.awt.*;

8   public class UserInterface implements Runnable {
9   
10      private JFrame frame;
11      private WordSequencer wordSequencer;
12      private FileOpener fileOpener;
13      private StatisticsRecorder stats;
14      private JLabel bigWord;
15  
16      public UserInterface(WordSequencer ws, FileOpener fo) {
17          wordSequencer = ws;
18          fileOpener = fo;
19      }
20  
21      /**
22       * Aloittaa uuden JFramen luonnin.
23       */
24      @Override
25      public void run() {
26          frame = new JFrame("FastReader");
27          frame.setPreferredSize(new Dimension(500, 300));
28  
29          frame.getContentPane().setBackground(Color.WHITE);
30  
31          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
32          frame.setResizable(false);
33          frame.setJMenuBar(createMenuBar());
34          createComponents(frame.getContentPane());
35  
36          frame.pack();
37          Dimension dimension = Toolkit.getDefaultToolkit().getScreenSize();
38          frame.setLocation((int)(dimension.getWidth() - frame.getWidth()) / 2,
39                          (int)(dimension.getHeight() - frame.getHeight()) / 2);
40          frame.setVisible(true);
41      }
42  
43      public void setBigWord(String word) {
44          bigWord.setText(word);
45      }
46  
47      private void createComponents(Container container) {
48          container.setLayout(new BoxLayout(container, BoxLayout.Y_AXIS));
49          container.add(createBigWord());
50          container.add(createControls());
51          container.add(Box.createVerticalGlue());
52          container.add(createSpeedSetter());
53      }
54  
55      private JPanel createSpeedSetter() {
56          JPanel speedPanel = new JPanel();
57          speedPanel.setBackground(Color.WHITE);
58          speedPanel.setOpaque(true);
59          speedPanel.setLayout(new BoxLayout(speedPanel, BoxLayout.X_AXIS));
60  
61          JLabel speedLabel = new JLabel("Speed: ");
62  
63          speedPanel.add(speedLabel);
64  
65          JTextField speedSetter = new JTextField();
66          speedSetter.addActionListener(new SpeedSetterActionListener(wordSequencer, speedSetter));
67          speedSetter.setMaximumSize(new Dimension(40, 14));
68          speedSetter.setMinimumSize(new Dimension(40, 14));
69          speedSetter.setBorder(BorderFactory.createMatteBorder(1, 1, 1, 1, Color.BLACK));
70          speedSetter.setText(wordSequencer.getSpeed() + "");
71  
72          speedPanel.add(speedSetter);
73  
74          return speedPanel;
75      }
76  
77      private JPanel createBigWord() {
78  
79          JPanel bigWordPanel = new JPanel();
80          bigWordPanel.setLayout(new BoxLayout(bigWordPanel, BoxLayout.X_AXIS));
81  
82          bigWord = new JLabel(wordSequencer.toBeginningOfText(), SwingConstants.CENTER);
83          bigWord.setAlignmentX(0.5f);
84          bigWord.setBorder(BorderFactory.createMatteBorder(2, 0, 2, 0, Color.BLACK));
85          bigWord.setMinimumSize(new Dimension(400, 100));
86          bigWord.setMaximumSize(new Dimension(400, 100));
87          bigWord.setFont(new Font("sans-serif", Font.PLAIN, 42));
88          bigWord.setBackground(Color.WHITE);
89          bigWord.setOpaque(true);
90  
91          bigWordPanel.add(bigWord);
92          return bigWordPanel;
93      }
94  
95      private JPanel createControls() {
96  
97          JPanel controlsPanel = new JPanel();
98          controlsPanel.setLayout(new BoxLayout(controlsPanel, BoxLayout.X_AXIS));
99          controlsPanel.setBackground(Color.WHITE);
100 
101         JButton[] buttons = new JButton[4];
102 
103         JButton rewindButton = new JButton(new ImageIcon("src/main/resources/rewind_button.png"));
104         rewindButton.addActionListener(new RewindActionListener(wordSequencer, bigWord));
105         buttons[0] = rewindButton;
106 
107         JButton prevButton = new JButton(new ImageIcon("src/main/resources/prev_sentence_button.png"));
108         prevButton.addActionListener(new BackwardsActionListener(wordSequencer, bigWord));
109         buttons[1] = prevButton;
110 
111         JButton playButton = new JButton(new ImageIcon("src/main/resources/play_button.png"));
112         playButton.addActionListener(new PlayActionListener(wordSequencer));
113         buttons[2] = playButton;
114 
115         JButton nextButton = new JButton(new ImageIcon("src/main/resources/next_sentence_button.png"));
116         nextButton.addActionListener(new ForwardsActionListener(wordSequencer, bigWord));
117         buttons[3] = nextButton;
118 
119         for (JButton jb : buttons) {
120             jb.setOpaque(false);
121             jb.setContentAreaFilled(false);
122             jb.setBorderPainted(false);
123             controlsPanel.add(jb);
124         }
125         return controlsPanel;
126     }
127 
128     private JMenuBar createMenuBar() {
129         JMenuBar menuBar = new JMenuBar();
130 
131         JMenuItem openFileItem = new JMenuItem("Open File");
132         openFileItem.addActionListener(new OpenFileActionListener(wordSequencer, fileOpener, bigWord));
133 
134         menuBar.add(openFileItem);
135 
136         JMenuItem pasteTextItem = new JMenuItem("Paste Text");
137         pasteTextItem.addActionListener(new PasteTextActionListener(wordSequencer, bigWord));
138 
139         menuBar.add(pasteTextItem);
140 
141         JMenuItem statisticsItem = new JMenuItem("Statistics");
142         statisticsItem.addActionListener(new StatisticsActionListener());
143 
144         menuBar.add(statisticsItem);
145 
146         return menuBar;
147     }
