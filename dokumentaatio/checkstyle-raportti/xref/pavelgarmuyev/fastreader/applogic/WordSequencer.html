8   public class WordSequencer {
9       private List<String> list;
10      private int speed, index;
11      private boolean running;
12      private StatisticsRecorder stats;
13      private FileOpener fileOpener;

15      /**
16       * Luo uuden WordSequencer -olion, joka navigoi tiedoston sanoissa ja lauseissa eteen- ja taaksepäin.
17       * @param statisticsPath    Polku tiedostolle, johon tallenetaan statistiikkaa.
18       */
19      public WordSequencer(String statisticsPath) {
20          stats = new StatisticsRecorder(statisticsPath);
21          fileOpener = new FileOpener();
22          list = Arrays.asList("Fastreader", "on", "ohjelma,", "joka", "auttaa", "sinua", "lukemaan", "nopeammin.", 
23                  "Kohdista", "katseesi", "rivin", "keskelle.", 
24                  "Mitä", "rennompi", "olet,", "sitä", "paremmin", "pystyt", "lukemaan.", 
25                  "Ruudun", "alalaidassa", "voit", "vaihtaa", "lukunopeutta.", 
26                  "Yläpalkissa", "olevasta", "Open", "File", "napista", "voit", "avata", "omia", "tiedostoja.", 
27                  "Statistics", "nappi", "avaa", "ikkunan,", "jossa", "näet", "kiinnostavaa", "tietoa", "lukutottumuksistasi.", 
28                  "Tiedot", "tallenetaan,", "jotta", "voit", "nähdä", "saavutuksesi", "ohjelman", "parissa.", 
29                  "Mukavia", "lukuhetkiä!");
30          index = 0;
31          speed = stats.getPreferredSpeed();
32          running = false;
33      }

62          return index;

64      /**
65       * Asettaa nykyisen sanan indeksin.
66       * Jos annettu parametri on alle 0, asettaa metodi indeksin nollakis.
67       * Jos annettu parametri on yli korkeimman indeksin, asettaa metodi indeksin korkeimmaksi mahdolliseksi.
68       * @param index     Asetettava indeksi.
69       */
70      public void setIndex(int index) {
71          if (index <= 0) {
72              this.index = 0;
73          } else if (index >= list.size() - 1) {
74              this.index = list.size() - 1;
75              setRunning(false);
76          } else {
77              this.index = index;
78          }
79      }
80      /**
81       * Asettaa indeksin yhden eteenpäin ja palauttaa uuden indeksin kohdalla olevan sanan.
82       * @return      Palauttaa seuraavan sanan.
83       */
84      public String nextWord() {
85          if (list.isEmpty()) {
86              return null;
87          }
88          setIndex(index + 1);
89          stats.incrementWordsRead();
90          return list.get(index);
91      }
92      /**
93       * Asettaa indeksin yhden taaksepäin ja palauttaa uuden indeksin kohdalla olevan sanan.
94       * @return      Palauttaa edellisen sanan.
95       */
96      public String prevWord() {
97          if (list.isEmpty()) {
98              return null;
99          }
100         setIndex(index - 1);
101         return list.get(index);
102     }
103     /**
104      * Nostaa indeksiä yhdellä, kunnes seuraava sana loppuu lopetusmerkkiin: . ! ?.
105      * Palauttaa tämän jälkeen seuraavan sanan, joka on seuraavan lauseen alku.
106      * @return     Seuraavan lauseen ensimmäinen sana.
107      */
108     public String nextSentenceBeginning() {
109         String word = list.get(index);
110         char c = word.charAt(word.length() - 1);
111         while (c != '.' && c != '!' && c != '?') {
112             setIndex(index + 1);
113             word = list.get(index);
114             c = word.charAt(word.length() - 1);
115         }
116         setIndex(index + 1);
117         return list.get(index);
118     }
119     /**
120      * Laskee indeksiä yhdellä, kunnes edellinen sana loppuu lopetusmerkkiin: . ! ?.
121      * Palauttaa tämän jälkeen seuraavan sanan.
122      * @return      Nykyisen lauseen ensimmäinen sana.
123      */
124     public String currentSentenceBeginning() {
125         if (index == 0) {
126             return list.get(index);
127         }
128         setIndex(index - 1);
129         String word = list.get(index);
130         char c = word.charAt(word.length() - 1);
131         boolean userWantsPrevSentence = (c == '.') || (c == '!') || (c == '?');
132 
133         if (userWantsPrevSentence) {
134             setIndex(index - 1);
135             word = list.get(index);
136             c = word.charAt(word.length() - 1);
137         }
138         while (c != '.' && c != '!' && c != '?') {
139             if (index == 0) {
140                 return list.get(index);
141             }
142             setIndex(index - 1);
143             word = list.get(index);
144             c = word.charAt(word.length() - 1);
145         }
146         setIndex(index + 1);
147         return list.get(index);
148     }
149     /**
150      * Asettaa indeksin 0 ja palauttaa indeksin kohdalla olevan sanan, joka on tekstin alku.
151      * @return      Tekstin ensimmäinen sana.
152      */
153     public String toBeginningOfText() {
154         index = 0;
155         return list.get(index);
156     }
157     /**
158      * Avaa tiedoston annetussa polussa.
159      * @param path  Avattavan tiedoston polku.
160      * @return      Palauttaa <code>true</code>, jos tiedosto on olemassa ja ei ole tyhjä. Muuten palauttaa <code>false</code>.
161      */
162     public boolean openPath(String path) {
163         List<String> newList = fileOpener.openFile(path);
164         if (newList == null) {
165             return false;
166         }
167         if (!newList.isEmpty()) {
168             list = newList;
169             return true;
170         }
171         return false;
172     }
173     /**
174      * Muokkaa HashMappiin tallenetut statistiikat luettavampaan muotoon;
175      * Isot alkukirjaimet,
176      * alaviivat välilyönneiksi,
177      * kaksoispiste ja välilyönti jakajana.
178      * Lisää perään muuttujan arvon ja rivinvaihdon, jotta kaikki muuttujat menisivät yhteen String olioon.
179      * @return  String olio, joka kuvaa nykyisten statistiikkaparametrien arvoja.
180      */
181     public String getStatistics() {
182         Map<String, String> statsMap = stats.getStatistics();
183         StringBuilder sb = new StringBuilder();
184         for (String key : statsMap.keySet()) {
185             char[] keyChars = key.toCharArray();
186             for (int i = 0; i < keyChars.length; i++) {
187                 if (i == 0) {
188                     sb.append((char) (keyChars[i] - 32));
189                 } else if (keyChars[i] == '_') {
190                     sb.append(' ');
191                 } else {
192                     sb.append(keyChars[i]);
193                 }
194             }
195             sb.append(": ").append(statsMap.get(key)).append("\n");
196         }
197         return sb.toString();
198     }
